
# ============================================================================
# GOTCHA — stale docker containers shadowing local dev servers
# ============================================================================
# `docker compose up --build` starts backend + frontend containers on the SAME
# ports (8000/5173) as local uvicorn / vite. Container images freeze the code
# at build time — they do NOT see your edits. If a container is still running
# from an earlier session, your local server can't bind the port and every
# request silently hits the old code (symptom: new routes 404 with
# {"detail": "Not Found"} while old routes work fine).
#
# For day-to-day dev: run ONLY the db in docker, backend/frontend locally:
#     docker compose up -d db
# Before starting local servers, check nothing is squatting on the ports:
#     docker compose ps                  # only db should be running
#     # PowerShell:
#     Get-NetTCPConnection -LocalPort 8000,5173 -State Listen -ErrorAction SilentlyContinue
# To stop stale containers:  docker compose down   (the pgdata volume survives)
# To verify which code is serving 8000, list its routes:
#     (Invoke-WebRequest http://localhost:8000/openapi.json).Content
# ============================================================================

# Project 4 ("Demo — Quality") is the M4 playground: 6 gold units expecting

export KEY=dev-admin-key
export A=1        # annotator id
export P=4        # quality demo project

# --- B1) Confirm the migration landed -------------------------------------
docker compose exec db psql -U minilp -d minilp -c \
  "\d+ labels" | grep gold_passed
docker compose exec db psql -U minilp -d minilp -c \
  "SELECT column_name FROM information_schema.columns \
   WHERE table_name='units' AND column_name IN ('quality','escalated_at');"

# --- B2) Server-side canonicalization: the client's `value` is ignored -----
# Lease a task, then submit a deliberately wrong `value`. The response must come
# back with the value the SERVER computed from `raw`.
SLOT=$(curl -s -H "Authorization: Bearer $KEY" \
  "localhost:8000/tasks/next?annotator=$A&project=$P" | python -c \
  "import sys,json;print(json.load(sys.stdin)['slot_id'])")
curl -s -X POST -H "Authorization: Bearer $KEY" -H 'content-type: application/json' \
  "localhost:8000/tasks/$SLOT/submit?annotator=$A" \
  -d '{"raw": {"category": "other:capybara"}, "value": {"category": "LIES"}}' \
  | python -m json.tool
#   -> "value": {"category": "capybara"}   (the "other:" prefix stripped, LIES ignored)
#   -> "quality": {paused, labels_voided, reputation, flags} and NOTHING about
#      gold pass/fail — golds must stay indistinguishable (§6.1).

# --- B3) Gold accuracy collapse pauses the annotator and voids their work ---
# Answer every task "dog". Six of the ten units are golds expecting "cat", so by
# the third graded gold the rolling accuracy is 0/3 and the pause fires.
for i in $(seq 1 8); do
  SLOT=$(curl -s -H "Authorization: Bearer $KEY" \
    "localhost:8000/tasks/next?annotator=$A&project=$P" | python -c \
    "import sys,json;d=sys.stdin.read();print(json.loads(d)['slot_id'] if d.strip() else '')" 2>/dev/null)
  [ -z "$SLOT" ] && break
  curl -s -X POST -H "Authorization: Bearer $KEY" -H 'content-type: application/json' \
    "localhost:8000/tasks/$SLOT/submit?annotator=$A" -d '{"raw": {"category": "dog"}}' \
    | python -c "import sys,json;q=json.load(sys.stdin)['quality'];print('paused:',q['paused'],'voided:',q['labels_voided'],'rep:',q['reputation'])"
done
#   -> the run ends with paused: True and a non-zero voided count.

# The annotator's report explains why (readable by the annotator themselves):
curl -s -H "Authorization: Bearer $KEY" localhost:8000/annotators/$A/report \
  | python -m json.tool
#   -> "status": "paused", "pause_reason": "gold accuracy 0.00 over last 3 golds...",
#      "live": {gold_accuracy, peer_agreement, variant_bias, speed_flags},
#      "events": [{"kind": "gold_fail", ...}, ...]

# Assignment is now refused with 403 + the reason (NOT a silent empty queue):
curl -s -o /dev/stderr -w '\nHTTP %{http_code}\n' -H "Authorization: Bearer $KEY" \
  "localhost:8000/tasks/next?annotator=$A&project=$P"
#   -> HTTP 403, detail: "gold accuracy ... is below threshold ..."

# Their labels are voided but KEPT (audit trail), and the slots reopened with
# their variant intact — this is the M4 balance guarantee:
docker compose exec db psql -U minilp -d minilp -c \
  "SELECT is_valid, count(*) FROM labels GROUP BY 1;"
docker compose exec db psql -U minilp -d minilp -c \
  "SELECT status, count(*) FROM slots s JOIN units u ON u.id=s.unit_id \
   WHERE u.project_id=4 GROUP BY 1;"
#   -> voided labels are is_valid=false; their slots are back to 'open'.

# Admin lifts the pause (already-voided work stays voided):
curl -s -X POST -H "Authorization: Bearer $KEY" \
  "localhost:8000/annotators/$A:resume" | python -m json.tool
curl -s -o /dev/null -w 'HTTP %{http_code}\n' -H "Authorization: Bearer $KEY" \
  "localhost:8000/tasks/next?annotator=$A&project=$P"     # -> HTTP 200 again

# --- B4) Dynamic overlap growth then escalation (§6.4) --------------------
# Needs two annotators disagreeing on the same unit. Create a second one:
docker compose exec db psql -U minilp -d minilp -c \
  "INSERT INTO users (email, role, created_at, updated_at) \
   VALUES ('second@local','annotator',now(),now()) ON CONFLICT DO NOTHING; \
   INSERT INTO annotators (kind, display_name, user_id, status, reputation_score, \
                           created_at, updated_at) \
   SELECT 'human','Second',id,'active',1.0,now(),now() FROM users \
   WHERE email='second@local';"
export B=$(docker compose exec -T db psql -U minilp -d minilp -tAc \
  "SELECT id FROM annotators WHERE display_name='Second';")

# Annotator A says "cat", annotator B says "dog", on the same unit.
for ANN in $A $B; do
  ANSWER=$([ "$ANN" = "$A" ] && echo cat || echo dog)
  SLOT=$(curl -s -H "Authorization: Bearer $KEY" \
    "localhost:8000/tasks/next?annotator=$ANN&project=$P" | python -c \
    "import sys,json;print(json.load(sys.stdin)['slot_id'])")
  curl -s -o /dev/null -X POST -H "Authorization: Bearer $KEY" \
    -H 'content-type: application/json' \
    "localhost:8000/tasks/$SLOT/submit?annotator=$ANN" -d "{\"raw\": {\"category\": \"$ANSWER\"}}"
done

# Per-unit consensus state, including the growth/escalation action:
curl -s -H "Authorization: Bearer $KEY" "localhost:8000/projects/$P/consensus" \
  | python -m json.tool
#   -> a disagreeing unit shows "action": "grown" and gains an extra open slot
#      (K=2 -> 3 -> 4); at max_labels_per_unit it flips to "action": "escalated"
#      with escalated_at set.

curl -s -H "Authorization: Bearer $KEY" "localhost:8000/projects/$P/consensus?escalated=true" \
  | python -c "import sys,json;print('escalated units:', json.load(sys.stdin)['count'])"

docker compose exec db psql -U minilp -d minilp -c \
  "SELECT id, status, escalated_at IS NOT NULL AS escalated, \
          quality->>'action' AS action, quality->'failed_keys' AS failed \
   FROM units WHERE project_id=4 ORDER BY id;"

# --- B5) Agreement analytics: kappa + entropy (§6.3) ----------------------
curl -s -H "Authorization: Bearer $KEY" \
  "localhost:8000/projects/$P/analytics/agreement" | python -m json.tool
#   -> per input key: kappa, observed_agreement, expected_agreement, method

#   Groups: ?group=human | model | cross  (cross is empty until judges land in M7)
curl -s -H "Authorization: Bearer $KEY" \

# --- B6) min_reputation gating (§6.2) -------------------------------------
# Create a project that demands a high score, then verify a fresh annotator is

#   - TSV upload (header + rows) and JSON-array upload both work via format=:
