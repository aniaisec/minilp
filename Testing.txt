#M1 Testing — Template engine + full data model

# --- Automated suite (runs against real PostgreSQL) ---
cd backend
pip install -e ".[dev]"
ruff check . && ruff format --check .
# No external DB needed locally: conftest spins up an in-process PostgreSQL
# (pgserver) and applies the Alembic migrations. To use your own DB instead:
#   export TEST_DATABASE_URL=postgresql+psycopg://minilp:minilp@localhost:5432/minilp_test
pytest -v            # 48 tests

# --- Manual verification against a live stack ---
docker compose up -d db                        # start PostgreSQL
cd backend
export MINILP_DATABASE_URL=postgresql+psycopg://minilp:minilp@localhost:5432/minilp
alembic upgrade head                           # create the full §4 schema
python -m app.seed                             # seed the 6 gallery templates
uvicorn app.main:app --reload                  # API at http://localhost:8000/docs

# 1) Gallery seeded (builtin, immutable) — expect 6 templates
curl -s localhost:8000/templates | python -m json.tool

# 2) Every gallery template round-trips validate -> preview
#    (image-classification is template id 2 after seeding)
curl -s -X POST localhost:8000/templates/2/preview \
  -H 'content-type: application/json' \
  -d '{"payload": {"image_url": "http://x/cat.png"}}' | python -m json.tool
#    -> payload_valid: true, inputs carry hotkey badges (1/2/3 + o for Other)

# 3) Clone a builtin -> editable custom draft; original untouched
curl -s -X POST localhost:8000/templates/2:clone \
  -H 'content-type: application/json' -d '{"new_name": "my-image-cls"}' | python -m json.tool

# 4) Duplicate/reserved hotkeys fail template save (expect HTTP 422 + errors)
curl -s -o /dev/stderr -w '%{http_code}\n' -X POST localhost:8000/templates \
  -H 'content-type: application/json' \
  -d '{"schema": {"name":"bad","inputs":[
        {"id":"a","type":"radio","label":"x","options":["p","q"],"hotkeys":["s","g"]}]}}'

# 5) Create a project on the flagship side-by-side template with overlap K=4
#    (side-by-side is template id 1). Non-divisible K is rejected (422).
curl -s -X POST localhost:8000/projects -H 'content-type: application/json' \
  -d '{"name":"pref","template_id":1,"labels_per_unit":4}' | python -m json.tool

# 6) Bulk-ingest units (JSONL). Malformed rows are rejected with row numbers;
#    valid rows are ingested and get K/n-balanced slots (2 AB + 2 BA each).
curl -s -X POST localhost:8000/projects/1/units:bulk -H 'content-type: application/json' \
  -d '{"jsonl": "{\"payload\": {\"prompt\": \"p1\", \"response_a\": \"a\", \"response_b\": \"b\"}}\n{bad json\n{\"payload\": {\"prompt\": \"p2\", \"response_a\": \"a\", \"response_b\": \"b\"}}"}' \
  | python -m json.tool
#    -> unit_count: 2, rejected_count: 1, rejected_rows: [{"row": 2, ...}]

# 7) Confirm exact per-variant slot balance in the DB (expect 2 AB + 2 BA per unit)
docker compose exec db psql -U minilp -d minilp -c \
  "SELECT u.id AS unit, s.variant->>'panel_order' AS variant, count(*) \
   FROM units u JOIN slots s ON s.unit_id=u.id GROUP BY 1,2 ORDER BY 1,2;"

# 8) Confirm the partial unique index exists (one valid label per annotator per unit)
docker compose exec db psql -U minilp -d minilp -c \
  "SELECT indexname FROM pg_indexes WHERE indexname='uq_labels_annotator_unit_valid';"
